1.Write a C program to write a character and display it using fgetc () and fputc () functions. 
                #include stdio.h
                int main()
                {
                 char ch;
                 FILE *file;
                 file = fopen("output.txt", "w");
                 if (file == NULL) return 1;
                 printf("Enter a character: ");
                 ch = getchar();
                 fputc(ch, file);
                 fclose(file);
                 file = fopen("output.txt", "r");
                 if (file == NULL) return 1;
                 char readChar = fgetc(file);
                 printf("Character read from file: %c\n", readChar);
                 fclose(file);
                 return 0;
                }

3. Write a C program to input a string and display it using fputs() and fgets() functions. 
                #include 
                int main() {
                char str[100];
                FILE *file;
                file = fopen("sample.txt", "w");
                if (file == NULL) return 1;
                printf("Enter a string: ");
                fgets(str, sizeof(str), stdin);
                fputs(str, file);
                fclose(file);
                file = fopen("sample.txt", "r");
                if (file == NULL) return 1;
                printf("String read from file: ");
                fgets(str, sizeof(str), file);
                printf("%s", str);
                fclose(file);
                return 0;
                }
4.Write a program to copy the contents of one file to another. 
                #include 
                int main() {
                FILE *source;
                source = fopen("sample.txt", "w");
                fprintf(source,"this is my file");
                fclose(source);
                FILE *destination;
                char ch;
                source = fopen("sample.txt", "r");
                if (source == NULL) {
                perror("Error opening source file");
                return 1;
                }
                destination = fopen("destination.txt", "w");
                if (destination == NULL) {
                perror("Error opening destination file");
                fclose(source);
                return 1;
                }
                while ((ch = fgetc(source)) != EOF) {
                fputc(ch, destination);
                }
                fclose(source);
                fclose(destination);
                printf("File copied successfully.\n");
                return 0;
                }

6.Write a C program to read character of 10th position and display it using fseek () and ftell() functions. 
                #include 
                int main() {
                FILE *source;
                source = fopen("sample.txt", "w");
                fprintf(source,"this is my file");
                fclose(source);
                FILE *file;
                char ch;
                file = fopen("sample.txt", "r");
                if (file == NULL) {
                perror("Error opening file");
                return 1;
                }
                fseek(file, 9, SEEK_SET);
                ch = fgetc(file);
                if (ch != EOF) {
                printf("Character at 10th position: %c\n", ch);
                } else {
                printf("There is no character at the 10th position.\n");
                }
                fclose(file);
                return 0;
                }
